-- Duplicate Country-Year Check

SELECT
    Entity,
    Year,
    COUNT(*) AS duplicates
FROM malaria
GROUP BY Entity, Year
HAVING COUNT(*) > 1;

-- Null Key Check

SELECT
    SUM(CASE WHEN Entity IS NULL THEN 1 ELSE 0 END) AS null_country,
    SUM(CASE WHEN Year IS NULL THEN 1 ELSE 0 END) AS null_year
FROM malaria;
